Connect to Supabase

Free database — no card needed. Takes 3 minutes to set up once.

✓ Your credentials are already saved below. Click Connect to Supabase to reconnect.
  1. Go to supabase.com → Start your project → sign up free with email.
  2. Click New project → name it smart-books → set a database password → choose a region → Create project. Wait ~1 minute.
  3. In the left sidebar click SQL Editor → click New query → paste the SQL below → click Run.
  4. Go to Project Settings → API → copy your Project URL and anon public key.
  5. Paste both below and click Connect.
SQL to paste in Step 3:

create table if not exists app_config(key text primary key, value text); create table if not exists users(id text primary key, name text, password text, role text, active boolean default true, created_at timestamptz default now()); create table if not exists products(id text primary key, name text, sku text, category text, cost_price numeric default 0, selling_price numeric default 0, quantity integer default 0, low_alert integer default 5, total_sold integer default 0, active boolean default true, created_at timestamptz default now()); create table if not exists sales(id text primary key, product_id text, product_name text, quantity integer, unit_price numeric, total numeric, payment_method text, customer text, notes text, sold_by text, sold_by_id text, created_at timestamptz default now()); create table if not exists restocks(id text primary key, product_id text, product_name text, quantity integer, cost_per_unit numeric, total_cost numeric, supplier text, notes text, recorded_by text, created_at timestamptz default now()); create table if not exists notifications(id text primary key, message text, type text, read_by text default '', created_at timestamptz default now()); create table if not exists devices(id text primary key, token text, authorized_at timestamptz default now(), ua text); create table if not exists suppliers(id text primary key, name text, contact_person text, phone text, email text, address text, notes text, active boolean default true, created_at timestamptz default now()); create table if not exists purchase_orders(id text primary key, supplier_id text, supplier_name text, items jsonb, total_cost numeric default 0, status text default 'pending', notes text, created_by text, created_at timestamptz default now());
